  "@context" : "http://library.stanford.edu/iiif/image-api/1.1/context.json",

  "height" : 4000,
  "scale_factors" : [ 1, 2, 4 ],

  "tile_height" : 1024,
  "formats" : [ "jpg", "png" ],
  "qualities" : [ "native", "grey" ],
